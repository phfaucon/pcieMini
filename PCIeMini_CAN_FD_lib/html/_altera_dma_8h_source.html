163  {
164  base[control_index] = ALTERA_AVALON_DMA_CONTROL_SOFTWARERESET_MSK;
165  base[control_index] = ALTERA_AVALON_DMA_CONTROL_SOFTWARERESET_MSK;

195  inline char* statusToString(char* buffer)
196  {
197  buffer[0] = 0;
198  uint32_t status = base[status_index];
199  if (status & ALTERA_AVALON_DMA_STATUS_DONE_MSK)
200  strcat(buffer, "DONE ");
201  if (status & ALTERA_AVALON_DMA_STATUS_BUSY_MSK)
202  strcat(buffer, "BUSY ");
203  if (status & ALTERA_AVALON_DMA_STATUS_REOP_MSK)
204  strcat(buffer, "RD end ");
205  if (status & ALTERA_AVALON_DMA_STATUS_WEOP_MSK)
206  strcat(buffer, "WR end ");
207  if (status & ALTERA_AVALON_DMA_STATUS_LEN_MSK)
208  strcat(buffer, "LEN=0 ");
209  return buffer;
210  }
211 
212  inline char* controlToString(char* buffer)
213  {
214  buffer[0] = 0;
215  uint32_t control = base[control_index];
216  if (control & ALTERA_AVALON_DMA_CONTROL_BYTE_MSK)
217  strcat(buffer, "8-bit ");
218  if (control & ALTERA_AVALON_DMA_CONTROL_HW_MSK)
219  strcat(buffer, "16-bit ");
220  if (control & ALTERA_AVALON_DMA_CONTROL_WORD_MSK)
221  strcat(buffer, "32-bit ");
222  if (control & ALTERA_AVALON_DMA_CONTROL_GO_MSK)
223  strcat(buffer, "GO ");
224  if (control & ALTERA_AVALON_DMA_CONTROL_I_EN_MSK)
225  strcat(buffer, "INT_EN ");
226  if (control & ALTERA_AVALON_DMA_CONTROL_REEN_MSK)
227  strcat(buffer, "RE_EN ");
228  if (control & ALTERA_AVALON_DMA_CONTROL_WEEN_MSK)
229  strcat(buffer, "WE_EN ");
230  if (control & ALTERA_AVALON_DMA_CONTROL_LEEN_MSK)
231  strcat(buffer, "Length_EN ");
232  if (control & ALTERA_AVALON_DMA_CONTROL_RCON_MSK)
233  strcat(buffer, "RCON ");
234  if (control & ALTERA_AVALON_DMA_CONTROL_WCON_MSK)
235  strcat(buffer, "WCON ");
236  if (control & ALTERA_AVALON_DMA_CONTROL_DWORD_MSK)
237  strcat(buffer, "64-bit ");
238  if (control & ALTERA_AVALON_DMA_CONTROL_QWORD_MSK)
239  strcat(buffer, "128-bit ");
240  return buffer;
241  }
