97 static inline int clock_gettime(int, struct timespec* spec) //C-file part
98 {
99  __int64 wintime;
100  GetSystemTimeAsFileTime((FILETIME*)&wintime);
101  wintime -= 116444736000000000i64; //1jan1601 to 1jan1970
102  spec->tv_sec = (long)(wintime / 10000000i64); //seconds
103  spec->tv_nsec = wintime % 10000000i64 * 100; //nano-seconds
104  return 0;
105 }

110 static inline void usleep(__int64 usec)
111 {
112  HANDLE timer;
113  LARGE_INTEGER ft;
114 
115  ft.QuadPart = -(10 * usec); // Convert to 100 nanosecond interval, negative value indicates relative time
116 
117  timer = CreateWaitableTimer(NULL, TRUE, NULL);
118  if (timer == NULL) return;
119  SetWaitableTimer(timer, &ft, 0, NULL, NULL, 0);
120  WaitForSingleObject(timer, INFINITE);
121  CloseHandle(timer);
122 }
