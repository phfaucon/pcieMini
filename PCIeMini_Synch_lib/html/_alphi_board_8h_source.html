101 static inline void usleep(__int64 usec)
102 {
103  HANDLE timer;
104  LARGE_INTEGER ft;
105 
106  ft.QuadPart = -(10 * usec); // Convert to 100 nanosecond interval, negative value indicates relative time
107 
108  timer = CreateWaitableTimer(NULL, TRUE, NULL);
109  if (timer == NULL) return;
110  SetWaitableTimer(timer, &ft, 0, NULL, NULL, 0);
111  WaitForSingleObject(timer, INFINITE);
112  CloseHandle(timer);
113 }
